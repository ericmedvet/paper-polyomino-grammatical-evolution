    ea.f.quality(of = ea.f.best(); format = "%6.2f");
    ea.f.hist(of = f.each(of = ea.f.all(); mapF = ea.f.quality()));

    f.gridW(of = ea.f.solution(of = ea.f.best()));
    f.gridH(of = ea.f.solution(of = ea.f.best()))
  ];
  kFunctions = [

)
